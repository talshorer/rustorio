3/// The tick is used to keep track of time in the game.
4/// You can advance the game by one tick using the [`next`](Tick::next) method.
5/// Many functions and building methods require a [`Tick`](Tick) to be passed in, which allows them to update their state.
6/// If a function takes a [`&mut Tick`](Tick), then the function will take time.
7/// If a function merely takes a [`&Tick`](Tick), or no [`Tick`](Tick) at all, it will never advance the game time.
8#[derive(Debug)]
9pub struct Tick {
10    tick: u64,
11    log: bool,
12}
13
14impl Tick {
15    pub(crate) fn start() -> Self {
16        Self { tick: 0, log: true }
17    }
18
19    /// Sets whether or not to log on tick advancement.
20    pub fn log(&mut self, log: bool) {
21        self.log = log;
22    }
23
24    /// Advances the game by one tick.
25    ///
26    /// By default prints the current tick number to the console.
27    /// If you want to disable this, use the [`log`](Tick::log) method.
28    pub fn advance(&mut self) {
29        self.advance_by(1);
30    }
31
32    /// Advances the game by the specified number of ticks.
33    ///
34    /// By default prints the current tick number to the console.
35    /// If you want to disable this, use the [`log`](Tick::log) method.
36    pub fn advance_by(&mut self, ticks: u64) {
37        self.tick = self.tick.checked_add(ticks).expect("Tick overflow. Well done you've found an exploit! Or you would have if `https://github.com/albertsgarde/rustorio/issues/3` hadn't beaten you to it!");
38        if self.log {
39            println!("{self}");
40        }
41    }
42
43    /// Advances the game until the specified tick number is reached.
44    /// Does nothing if the target tick is less than or equal to the current tick.
45    ///
46    /// By default prints the current tick number to the console.
47    /// If you want to disable this, use the [`log`](Tick::log) method.
48    pub fn advance_to_tick(&mut self, target_tick: u64) {
49        if target_tick > self.tick {
50            self.advance_by(target_tick - self.tick);
51        }
52    }
53
54    /// Advances the game until the specified condition is met or the maximum number of ticks has passed.
55    /// Returns `true` if the condition was met, or `false` if the maximum number of ticks was reached first.
56    ///
57    /// By default prints the current tick number to the console every tick.
58    /// If you want to disable this, use the [`log`](Tick::log) method.
59    pub fn advance_until<F>(&mut self, mut condition: F, max_ticks: u64) -> bool
60    where
61        F: FnMut(&Tick) -> bool,
62    {
63        let start_tick = self.tick;
64        while !condition(self) && self.tick - start_tick < max_ticks {
65            self.advance();
66        }
67        self.tick - start_tick < max_ticks
68    }
69
70    /// Returns the current tick number.
71    pub fn cur(&self) -> u64 {
72        self.tick
73    }
74}
75
76impl From<&Tick> for u64 {
77    fn from(tick: &Tick) -> Self {
78        tick.tick
79    }
80}

82impl PartialOrd<u64> for &Tick {
83    fn partial_cmp(&self, other: &u64) -> Option<std::cmp::Ordering> {
84        Some(self.tick.cmp(other))
85    }
86}
87
88impl PartialOrd<&Tick> for u64 {
89    fn partial_cmp(&self, other: &&Tick) -> Option<std::cmp::Ordering> {
90        Some(self.cmp(&other.tick))
91    }
92}
93
94impl PartialEq<u64> for &Tick {
95    fn eq(&self, other: &u64) -> bool {
96        self.tick == *other
97    }
98}
99
100impl PartialEq<&Tick> for u64 {
101    fn eq(&self, other: &&Tick) -> bool {
102        *self == other.tick
103    }
104}
