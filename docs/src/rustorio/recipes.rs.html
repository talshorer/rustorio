1//! A recipe is a way of turning resources into other resources.
2//! A specific recipe specifies the input and output resources, as well as the time it takes to complete the recipe.
3
4use std::fmt::Debug;
5
6use crate::ResourceType;
7
8/// Any recipe that implements this trait can be used in an [`Assembler`](crate::buildings::Assembler).
9pub trait AssemblerRecipe: Debug {
10    /// The first of the two types of resources consumed by this recipe.
11    const INPUT1: ResourceType;
12    /// The amount of the first input resource consumed by this recipe.
13    const INPUT1_AMOUNT: u32;
14    /// The second of the two types of resources consumed by this recipe.
15    const INPUT2: ResourceType;
16    /// The amount of the second input resource consumed by this recipe.
17    const INPUT2_AMOUNT: u32;
18    /// The type of resource produced by this recipe.
19    const OUTPUT: ResourceType;
20    /// The amount of the output resource produced by this recipe.
21    const OUTPUT_AMOUNT: u32;
22    /// The time (in ticks) it takes to complete this recipe.
23    const TIME: u64;
24}
25
26/// The recipe you need to win! An [`Assembler`](crate::buildings::Assembler) recipe that creates points. Converts 4 iron and 4 copper into 1 point resource. Takes 20 ticks.
27#[derive(Debug)]
28pub struct PointRecipe;
29
30impl AssemblerRecipe for PointRecipe {
31    const INPUT1: ResourceType = ResourceType::Iron;
32    const INPUT1_AMOUNT: u32 = 4;
33    const INPUT2: ResourceType = ResourceType::Copper;
34    const INPUT2_AMOUNT: u32 = 4;
35    const OUTPUT: ResourceType = ResourceType::Point;
36    const OUTPUT_AMOUNT: u32 = 1;
37    const TIME: u64 = 20;
38}

40/// Any recipe that implements this trait can be used in a [`Furnace`](crate::buildings::Furnace).
41pub trait FurnaceRecipe: Debug {
42    const INPUT: ResourceType;
43    const INPUT_AMOUNT: u32;
44    const OUTPUT: ResourceType;
45    const OUTPUT_AMOUNT: u32;
46    const TIME: u64;
47}
