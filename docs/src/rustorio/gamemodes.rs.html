7/// A game mode defines the starting resources and victory conditions for a game.
8pub trait GameMode {
9    /// Starting resources provided to the player at the beginning of the game.
10    #[allow(private_bounds)]
11    type StartingResources: StartingResources;
12    /// Resources required to achieve victory.
13    type VictoryResources;
14}
15
16/// Starting resources for the tutorial game mode.
17pub struct TutorialStartingResources {
18    pub iron: Bundle<{ ResourceType::Iron }, 10>,
19}
20
21impl StartingResources for TutorialStartingResources {
22    fn init() -> Self {
23        Self { iron: Bundle::new() }
24    }
25}
26
27/// The tutorial game mode. Very short distance from start to victory, meant to teach the very most basic elements of play.
28pub struct Tutorial;
29
30impl GameMode for Tutorial {
31    type StartingResources = TutorialStartingResources;

33}
