1//! A recipe is a way of turning resources into other resources.
2//! A specific recipe specifies the input and output resources, as well as the time it takes to complete the recipe.
3
4use std::{fmt::Debug, marker::PhantomData};
5
6use crate::ResourceType;
7
8macro_rules! create_recipe {
9    ($recipe_name:ident) => {
10        #[derive(Debug, Clone, Copy)]
11        struct $recipe_name {
12            pub(crate) dummy: (),
13        }
14    };
15}
16
17/// Any recipe that implements this trait can be used in an [`Assembler`](crate::buildings::Assembler).
18pub trait AssemblerRecipe: Debug {
19    const INPUT1: ResourceType;
20    const INPUT1_AMOUNT: u32;
21    const INPUT2: ResourceType;
22    const INPUT2_AMOUNT: u32;
23    const OUTPUT: ResourceType;
24    const OUTPUT_AMOUNT: u32;
25    const TIME: u64;
26}
27
28/// The recipe you need to win! An [`Assembler`](crate::buildings::Assembler) recipe that creates points. Converts 4 iron and 4 copper into 1 point resource. Takes 20 ticks.
29create_recipe! {PointRecipe}
30
31impl AssemblerRecipe for PointRecipe {
32    const INPUT1: ResourceType = ResourceType::Iron;
33    const INPUT1_AMOUNT: u32 = 4;
34    const INPUT2: ResourceType = ResourceType::Copper;
35    const INPUT2_AMOUNT: u32 = 4;
36    const OUTPUT: ResourceType = ResourceType::Point;
37    const OUTPUT_AMOUNT: u32 = 1;
38    const TIME: u64 = 20;
39}

41/// Any recipe that implements this trait can be used in a [`Furnace`](crate::buildings::Furnace).
